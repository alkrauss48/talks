mutation AddItemMutation(
  $title: String!
  $description: String
  $imageUrl: String
) {
  addItem(input: {
    title: $title, description: $description, imageUrl: $imageUrl
  }) {
    item {
      id
      title
      description
      imageUrl
      user {
        id
        fullName
      }
    }
  }
}
